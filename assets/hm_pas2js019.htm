   s : array [0..1] of Float;

j := 1;

d[i] += 10;
d[j] += 100;
s[i] += 10;
s[j] += 100;

s[i] -= 1;
s[j] -= 2;

d[i] *= 2;
d[j] *= 3;
s[i] *= 2;
s[j] *= 3;

s[i] /= 2;
s[j] /= 4;

WriteLn(d[0]);   // 20
WriteLn(d[1]);   // 300
WriteLn('--------------');
