Example code : UpperCase, LowerCase, ToLower, ToUpper

procedure TForm1.W3Button15Click(Sender: TObject);

var abc, lower, upper : string;

 

begin

// Convert a lower case letter to upper case

lower := 'g';

upper := UpperCase(lower);

 

// Display these letters

WriteLn('lower = '+lower+' upper = '+upper);

 

abc := "How are you doing today?;I'm fine, and you?";

WriteLn(abc.ToLower);

WriteLn(LowerCase(abc));

 

WriteLn(abc.ToUpper);

WriteLn(UpperCase(abc));

 

end;
