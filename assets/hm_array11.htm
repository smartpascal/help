Example code : Declaring and instantiate an array in SmartMS

// Defining an array

var

  a : array of Integer;

  i : Integer;

begin

// Instantiate an array

 a := new Integer[10];

 for i := a.Low to a.High do begin

  a[i] := i;

  WriteLn(a[i]);

 end;
