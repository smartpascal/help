procedure sss(const arr: array of const);
begin
  WriteLn(Length(arr));
  WriteLn(': ');
  var   i: Integer;
  for i := 0 to Length(arr) - 1 do
  begin
    WriteLn(arr[i]);
    WriteLn(',');
  end;
  WriteLn('');
end;

procedure sss2(const arr: array of const);
