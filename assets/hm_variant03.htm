Using Variants to create javascript object with ASM

 Top 

Example code : storing object values using ASM section

procedure TForm1.W3Button2Click(Sender: TObject);

var mData: variant;

Begin

 asm @mData = {}; end; // Set variant to "empty JS object"

 

 mData.miName := 'warleyalex';

 mData.miAge := 9;

 mData.miGender := 1;

  

 WriteLn(mData);

end;

 

{

    miName: "warleyalex", 

    miAge: 9, 

    miGender: 1

 }
